query{
  getUser(id:"Titletest"){
    Id
    name
    surname
    address
    email
    publishDate
  }
}


#Vypsani vsech dokumentu v dbs
query{
  users{
    Id
    name
    surname
    address
    email
    publishDate
  }
}

#Vytvoreni dokumentu v dbs
mutation{
  createUser(userC:{Id:"PrvkyNavic",name:"Titletest",surname:"insTest",address:"Adresa",email:"emailUz@email.cz"}){
    result{
      Id
      name
      surname
      address
      email
      publishDate
    }
  }
}

#Vytvoreni group dokumentu v dbs

  createGroup(groupC:{Id:"SkupinaID",name:"TestovaciSkupina"}){
